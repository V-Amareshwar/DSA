//  InPostOrder in Binary Tree

import java.util.*;
class Main{
    public static void main(String[] args){
        int arr [] = {1,2,3,4,5,6,7};
        Node root = Create( arr, 0);
        PostOrder(root);
    }
    public static Node Create(int arr[] , int i){
        if(i>=arr.length){
            return null;
        }
        Node root = new Node(arr[i]);
        root.left = Create(arr,2*i+1);
        root.right = Create(arr,2*i+2);
        return root;
    }
    public static void PostOrder(Node root){
        if(root == null){
            return;
        }
    
        PostOrder(root.left);
        PostOrder(root.right);
        System.out.print(root.data+" ");
    }
}

class Node{
    int data;
    Node left,right;
    Node(int data){
        this.data = data;
        left = right = null;
    }
}
